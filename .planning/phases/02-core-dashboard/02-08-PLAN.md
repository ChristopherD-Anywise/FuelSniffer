Invite code generation (RESEARCH.md "Don't Hand-Roll" section):
- `crypto.randomBytes(4).toString('hex')` = 8-char hex string (e.g. "a3f82b9c")
- Short enough to type manually; adequate entropy for a friend group

Admin route access control: These routes are server-to-server (owner only, no UI in Phase 2). For Phase 2, protect with the same session cookie as the dashboard. A more robust admin token can be added in a future phase if needed. Check req.cookies.get('session') in each handler and return 401 if no valid session.

Cloudflare Tunnel (cloudflared):
- No API key needed for a quick tunnel: `cloudflared tunnel --url http://app:3000`
- Gives a randomly-assigned https://*.trycloudflare.com URL valid for the session
- For a persistent named tunnel (e.g. fuelsniffer.example.com), owner must create the tunnel in Cloudflare dashboard first and provide a tunnel token
- For Phase 2, the ephemeral URL approach is sufficient for sharing with friends during testing

import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/lib/db/client'
import { inviteCodes } from '@/lib/db/schema'
import { decrypt } from '@/lib/session'
import { eq } from 'drizzle-orm'
import { randomBytes } from 'crypto'
import { z } from 'zod'

// Auth check helper — returns null if session is invalid
async function requireSession(req: NextRequest) {
  const cookie = req.cookies.get('session')?.value
  return decrypt(cookie)
}

/**
 * GET /api/admin/invite-codes
 * Returns all invite codes (for owner to see who has access).
 */
export async function GET(req: NextRequest) {
  const session = await requireSession(req)
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const codes = await db.select().from(inviteCodes).orderBy(inviteCodes.createdAt)
  return NextResponse.json(codes, { status: 200 })
}

const CreateSchema = z.object({
  label: z.string().min(1).max(100).optional(),
})

/**
 * POST /api/admin/invite-codes
 * Body: { label?: string }
 * Creates a new invite code and returns it.
 */
export async function POST(req: NextRequest) {
  const session = await requireSession(req)
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const body = await req.json().catch(() => ({}))
  const parsed = CreateSchema.safeParse(body)
  if (!parsed.success) {
    return NextResponse.json({ error: 'Invalid request body' }, { status: 400 })
  }

  const code = randomBytes(4).toString('hex')  // 8-char hex e.g. "a3f82b9c"

  const [created] = await db
    .insert(inviteCodes)
    .values({ code, label: parsed.data.label ?? null })
    .returning()

  return NextResponse.json(created, { status: 201 })
}

const DeleteSchema = z.object({
  id: z.string().regex(/^\d+$/).transform(Number),
})

/**
 * DELETE /api/admin/invite-codes?id=N
 * Soft-deletes (sets is_active=false) the specified code.
 * Per D-13: revocation must be per-code and non-destructive.
 */
export async function DELETE(req: NextRequest) {
  const session = await requireSession(req)
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const parsed = DeleteSchema.safeParse({ id: req.nextUrl.searchParams.get('id') })
  if (!parsed.success) {
    return NextResponse.json({ error: 'id query param is required and must be a positive integer' }, { status: 400 })
  }

  await db
    .update(inviteCodes)
    .set({ isActive: false })
    .where(eq(inviteCodes.id, parsed.data.id))

  return NextResponse.json({ ok: true }, { status: 200 })
}

Read fuelsniffer/docker-compose.yml first. Then add a `cloudflared` service to enable external access.

The Cloudflare Tunnel setup:
- If Caddy is already in docker-compose.yml, the tunnel points to Caddy's port (e.g. http://caddy:80)
- If there is no Caddy service yet (it may be deferred), the tunnel points directly to the Next.js app: http://app:3000 (where 'app' is the Next.js service name in the compose file)

Add this service to the services block:

```yaml
  cloudflared:
    image: cloudflare/cloudflared:latest
    restart: unless-stopped
    command: tunnel --no-autoupdate --url http://app:3000
    # For a persistent named tunnel (recommended for production):
    # 1. Create tunnel: cloudflared tunnel create fuelsniffer
    # 2. Set CLOUDFLARE_TUNNEL_TOKEN in .env
    # 3. Replace command with: tunnel run --token ${CLOUDFLARE_TUNNEL_TOKEN}
    depends_on:
      - app
    networks:
      - fuelsniffer
```

Key notes:
1. `--url http://app:3000` creates an ephemeral trycloudflare.com URL — no Cloudflare account needed, URL changes on restart
2. The service name `app` must match the Next.js service name in the existing docker-compose.yml — check and adjust if different
3. If there is no `networks` block in the existing file, use the default Docker network (remove the networks lines)
4. The tunnel URL is printed to the cloudflared container logs: `docker compose logs cloudflared`

If Caddy is present in docker-compose.yml, point the tunnel at Caddy's service instead:
`command: tunnel --no-autoupdate --url http://caddy:80`
